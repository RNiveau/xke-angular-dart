  LogController(Http http) {
    logs = [];
      Future.wait([http.get("apache-log.json").then((HttpResponse httpResponse) {
      for (Map log in httpResponse.data) {
        Log l = new Log.fromJsonMap(log);
        logs.add(l);
      }
    })]);
  }
